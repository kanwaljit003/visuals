# 📦 Import required libraries
import pandas as pd

# 📄 Load the datasets
indicator_1 = pd.read_csv("unicef_indicator_1 (1).csv")
metadata = pd.read_csv("unicef_metadata (1).csv")

# 🔁 Merge indicator_1 with metadata using 'country' and 'time_period' == 'year'
merged_df = pd.merge(
    indicator_1,
    metadata,
    how="left",
    left_on=["country", "time_period"],
    right_on=["country", "year"]
)

# 🧹 Drop unnecessary columns (like duplicate alpha/numeric codes)
columns_to_drop = ['alpha_2_code_y', 'alpha_3_code_y', 'numeric_code_y']
merged_df = merged_df.drop(columns=columns_to_drop, errors='ignore')

# 🏷️ Rename some columns for clarity
merged_df = merged_df.rename(columns={
    "alpha_2_code_x": "alpha_2_code",
    "alpha_3_code_x": "alpha_3_code",
    "numeric_code_x": "numeric_code"
})

import geopandas as gpd
import matplotlib.pyplot as plt
import pandas as pd

# Load shapefile
world = gpd.read_file("ne_110m_admin_0_countries.shp")

# 🧮 Group by country and take average over all years
avg_df = merged_df.dropna(subset=["obs_value"]) \
                  .groupby(["alpha_3_code", "country"], as_index=False)["obs_value"].mean()

# 🌍 Merge with world shapefile using ISO codes
world_avg_map = world.merge(avg_df, how="left", left_on="ISO_A3", right_on="alpha_3_code")

# 🎨 Plot the final map
fig, ax = plt.subplots(1, 1, figsize=(16, 9))

world_avg_map.plot(
    column='obs_value',
    ax=ax,
    legend=True,
    cmap='YlGnBu',
    edgecolor='black',
    linewidth=0.4,
    missing_kwds={
        "color": "lightgrey",
        "edgecolor": "white",
        "hatch": "///",
        "label": "No Data"
    }
)

# ✏️ Title and formatting
ax.set_title("Average Essential Health Services Coverage (2000–2017)", fontsize=18, weight='bold')
ax.axis('off')

# 🔧 Tweak legend
leg = ax.get_legend()
if leg:
    leg.set_bbox_to_anchor((1.02, 0.5))
    leg.set_title("Coverage (%)")

plt.tight_layout()
plt.show()
